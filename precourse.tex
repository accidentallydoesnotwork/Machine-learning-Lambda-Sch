Mv =

\begin{pmatrix}
m_{1,1} \  m_{1,2}
\\ m_{2,1} \  m_{2,2}
\end{pmatrix}

\begin{pmatrix}
v_1
\\ v_2
\end{pmatrix}

=

\begin{pmatrix}
v_1 \times m_{1,1} + v_2 \times m_{1,2}
\\ v_1 \times m_{2,1} + v_2 \times m_{2,2}
\end{pmatrix}
